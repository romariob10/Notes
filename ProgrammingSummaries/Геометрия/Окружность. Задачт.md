#### №1 Прямая и окружность
```cpp
#include <cmath>
#include <iomanip>
#include <iostream>
#include <vector>
using namespace std;

int main() {
  double x0, y0, r, a, b, c;
  cin >> x0 >> y0 >> r >> a >> b >> c;
  
  double alfa = abs(a * x0 + b * y0 + c) / (a * a + b * b);
  double h = abs(a * x0 + b * y0 + c) / sqrt(a * a + b * b);
  
  double x = x0 + alfa * a, y = y0 + alfa * b;

  if (a * x + b * y + c > 0.00001) {
    x = x0 - alfa * a;
    y = y0 - alfa * b;
  }

  if (h - r > 0.00001) {
    cout << 0;
    return 0;
  } else if (abs(h - r) == 0.00001 || (h - r) == 0) {
    cout << fixed << setprecision(5) << 1 << endl << x << ' ' << y;
    return 0;
  }

  double n = sqrt(r * r - h * h);
  double beta = n / sqrt(a * a + b * b);
  
  cout << fixed << setprecision(5) << 2 << endl << x - beta * b << ' ' << y + beta * a << endl;
  cout << fixed << setprecision(5) << x + beta * b << ' ' << y - beta * a;
}
```
