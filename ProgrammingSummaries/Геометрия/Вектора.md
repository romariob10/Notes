```C++
#include <cmath>
#include <iostream>
#include <vector>

using namespace std;

class Vector {
 public:
  int x = 0, y = 0;
  Vector() { cin >> x >> y; }
  Vector(int x, int y) : x(x), y(y) {}
  Vector(int x1, int y1, int x2, int y2) : x(x2 - x1), y(y2 - y1) {}

  Vector operator+(const Vector& v) const { return Vector(x + v.x, y + v.y); }

  Vector operator-(const Vector& v) const { return Vector(x - v.x, y - v.y); }

  int operator*(const Vector& v) const {  // Сколярное
    return x * v.x + y * v.y;
  }

  int operator^(const Vector& v) const {  // Векторное
    return x * v.y - y * v.x;
  }
  double length() const { return sqrt(double(x * x + y * y)); }

  double length(const Vector& v) const {
    return sqrt(double((x - v.x) * (x - v.x) + (y - v.y) * (y - v.y)));
  }
};
```
